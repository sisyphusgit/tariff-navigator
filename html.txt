        function switchTab(tab) {
            document.querySelectorAll('.tab-content').forEach(el => el.classList.remove('active'));
            document.querySelectorAll('.tab-btn').forEach(el => el.classList.remove('active'));
            document.getElementById(tab).classList.add('active');
            event.target.classList.add('active');
        }

        function updateProducts() {
            const category = document.getElementById('tariffCategory').value;
            const select = document.getElementById('tariffProduct');
            select.innerHTML = '<option value="">-- Select Specific Product --</option>';
            if (category && tariffDatabase[category]) {
                tariffDatabase[category].forEach(p => {
                    const opt = document.createElement('option');
                    opt.value = JSON.stringify(p);
                    opt.textContent = p.name;
                    select.appendChild(opt);
                });
            }
        }

        function findTariff() {
            const category = document.getElementById('tariffCategory').value;
            const productStr = document.getElementById('tariffProduct').value;
            if (!category || !productStr) {
                alert('Please select both category and product');
                return;
            }
            const product = JSON.parse(productStr);
            const resultsDiv = document.getElementById('tariffResults');
            resultsDiv.innerHTML = `
                <div class="result-card">
                    <h4>${product.name}</h4>
                    <div>HS Code: <span class="hs-code">${product.code}</span></div>
                    <div class="rate-box">
                        <strong>Current US Tariff Rate:</strong> ${product.us}<br>
                        <strong>USMCA Rate (if qualified):</strong> ${product.usmca}
                    </div>
                    <p style="margin-top: 15px; color: #666; font-size: 0.9em;">

                    </p>
                </div>

        }

        function findPrograms() {
            const province = document.getElementById('progProvince').value;
            const need = document.getElementById('progNeed').value;
            if (!province || !need) {
                alert('Please select province and what you need');
                return;
            }
            const resultsDiv = document.getElementById('programResults');
            const programs = programDatabase[province][need] || [];
            let html = '';
            if (programs.length > 0) {
                programs.forEach(prog => {
                    html += `
                        <div class="program-card">
                            <h4>${prog.name}</h4>
                            <div class="amount">${prog.amount}</div>
                            <p>${prog.desc}</p>
                            <span class="match-score">‚úì You Likely Qualify</span>

            }

            resultsDiv.classList.add('active');
        }

        function generatePlan() {
            const company = document.getElementById('companyName').value;
            const province = document.getElementById('compProvince').value;
            const industry = document.getElementById('compIndustry').value;
            const imports = parseFloat(document.getElementById('annualImports').value);
            const employees = document.getElementById('employees').value;
            const margin = parseFloat(document.getElementById('margin').value);
            
            if (!company || !province || !industry || !imports || !employees || !margin) {
                alert('Please complete all fields');
                return;
            }

            let tariffRate = 0.15;
            if (industry === 'steel') tariffRate = 0.25;
            if (industry === 'auto') tariffRate = 0.20;

            const tariffCost = imports * tariffRate;
            const currentProfit = imports * (margin / 100);
            const profitAfter = currentProfit - tariffCost;
            const impactPercent = ((profitAfter - currentProfit) / currentProfit * 100).toFixed(1);

            let programs = [];
            if (employees !== '250+') {
                programs.push({ name: 'RTRI Grants', amount: '$1M' });
                programs.push({ name: 'CanExport SME', amount: '$75K' });
            }
            if (imports > 500000) {
                programs.push({ name: 'BDC Tariff Loan', amount: '$5M' });
            }

            let html = `

                    <p><strong>Impact on Profit:</strong> <span style="color: #dc2626; font-weight: 600;">${impactPercent}%</span></p>
